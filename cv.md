#### ***Fill an array of 50 elements with odd numbers from 1 to 99. (Use the remainder-of-division operation to check for odd numbers.)***
```
#include <iostream>
#include <ctime>
#include <cstdlib>
using namespace std;
int main(){
	const int SIZE = 50;
    int ourArr[SIZE] = {};
    srand(time(NULL));
    for (int i = 0; i < SIZE; ){
		ourArr[i] = 1 + rand() % 100;
		if (ourArr[i] % 2 != 0){
			cout << ourArr[i] << " | ";
			i++; }
	}
	cout << endl << endl;
	return 0; }
```
